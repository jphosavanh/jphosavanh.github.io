def server(input, output, session):
  
    @reactive.calc
    def staff_file():
        file: list[FileInfo] | None = input.staff_details()
        if file is None:
            return None
        df = pd.read_csv(file[0]['datapath'])
        # df.columns = ['Name', 'PhDStatus', 'Role']
        df.set_index('Name', inplace=True)
        return df    
    
    @reactive.calc
    def guess_code():
        file: list[FileInfo] | None = input.staff_details()
        if file is None:
            return None
        possible = file[0]['name'][:8]
        if re.match('[a-zA-Z]{4}[0-9]{4}', possible):
            return possible.upper()
        else:
            return None
    
    @render.ui
    def ui_staff_list_quick_select():
        if staff_file() is not None:
            return ui.input_checkbox_group(  
                'roles_selected',  
                'Staff (Quick select)',  
                {  
                    i: i for i in staff_file()['Role'].unique()
                } | {'Unselect all' : 'Unselect all'},
            ),
    
    @render.ui
    @reactive.event(input.roles_selected, ignore_none=False)
    def ui_staff_list():
        if staff_file() is not None:
            selected = [i for i in staff_file()[staff_file()['Role'].isin(input.roles_selected())].index]
            if 'Unselect all' in input.roles_selected():
                selected = []
            return ui.input_checkbox_group(  
                'staff_selected',  
                'Staff',  
                {  
                    i: i for i in staff_file().index
                },
                selected=selected
            ),
    
    @render.ui
    def ui_unit():
        return ui.input_text(
            'unit', 'UoS Code', value=guess_code()
        ),
        
    @render.ui
    @reactive.event(input.activity)
    def ui_other_activity():  
        if input.activity() == 'other_activity':
            return  ui.input_text(
                'other_activity', 'Other activity'
            )
        
    @render.ui
    @reactive.event(input.variation)
    def ui_other_reason():  
        if input.variation() == 'other_reason':
            return  ui.input_text(
                'other_reason', 'Other variation reason'
            )
        
    @render.ui
    @reactive.event(input.scheduled)
    def ui_scheduled_start_time():  
        if input.scheduled() == 'Y':
            return ui.input_text(
                'scheduled_start_time', 'Start time (24-hour HHMM)'
            )
        
    @render.ui
    @reactive.event(input.scheduled)
    def ui_scheduled_start_date():  
        if input.scheduled() == 'Y':
            return ui.input_date(
                'scheduled_start_date', 'Start date:', format='dd/mm/yy'
            )
        
    @render.ui
    @reactive.event(input.scheduled)
    def ui_scheduled_weeks():  
        if input.scheduled() == 'Y':
            return ui.input_text(
                'scheduled_weeks', 'Weeks'
            )

    @render.ui
    @reactive.event(input.scheduled)
    def ui_flex():  
        if input.scheduled() == 'N':
            return ui.input_radio_buttons(  
                'flex',  
                'Flex-time',  
                {'Y': 'Yes', 'N': 'No'},  
            ),
        
    @render.ui
    @reactive.event(input.flex)
    def ui_unscheduled_start_date():  
        if input.flex() == 'N':
            return ui.input_date(
                'unscheduled_start_date', 'Start date:', format='dd/mm/yy'
            ),
        
    @render.ui
    @reactive.event(input.flex)
    def ui_unscheduled_weeks():  
        if input.flex() == 'N':
            return ui.input_text(
                'unscheduled_weeks', 'Weeks'
            ),  
  
    @render.text
    def get_date():
        return input.start_date()

    current_df = reactive.value(base_df)
    rows_added = reactive.value(0)

    @reactive.effect    
    @reactive.event(input.add_button)
    def _():
        current_df.set(render_df.data_view())
        to_add = []
        
        for name in input.staff_selected():

            phd_stat = staff_file().loc[name, 'PhDStatus']

            if input.role() == 'Coordinator':
                phd_stat = 'y'

            if input.activity() == 'other_activity':
                activity = input.other_activity()
            else:
                activity = input.activity()

            if input.variation() == 'other_reason':
                variation = input.other_reason()
            else:
                variation = input.variation()

            if input.role() == 'Use prefill':
                role = staff_file().loc[name, 'Role']
            else:
                role = input.role()

            if input.activity() == 'MARKING':
                code = 'M05'
            elif input.activity() == 'tut':
                activity = 'TEACHING (Tutorial)'
                if role == 'Demonstrator':
                    code = 'DE2'
                else:
                    code = 'TU2'
            elif input.activity() == 'retut':
                activity = 'TEACHING (Repeat tutorial)'
                if role == 'Demonstrator':
                    code = 'DE2'
                else:
                    code = 'TU4'
            elif input.activity() == 'dem':
                activity = 'TEACHING (demonstrating)'
                code = 'DE2'
            elif input.activity() == 'lec':
                activity = 'TEACHING (lecture)'
                code = ''
            elif input.activity() == 'other_activity':
                code = ''
            else:
                code = 'A02'
            
            if phd_stat.lower() == 'y' and len(code) > 1:
                code = code[:-1] + str(int(code[-1]) - 1)

            if input.scheduled() == 'Y':
                unscheduled_flex = 'N'
                unscheduled_start = 'NA'
                unscheduled_weeks = 'NA'
                scheduled_start_time = input.scheduled_start_time()
                scheduled_start = input.scheduled_start_date().strftime('%d %b %Y')
                scheduled_weeks = input.scheduled_weeks()

            else:
                scheduled_start_time = ''
                scheduled_start = ''
                scheduled_weeks = ''
                if input.flex() == 'Y':
                    unscheduled_flex = 'Y'
                    unscheduled_start = 'NA'
                    unscheduled_weeks = 'NA'
                else:
                    unscheduled_flex = 'N'
                    unscheduled_start = input.unscheduled_start_date().strftime('%d %b %Y')
                    unscheduled_weeks = input.unscheduled_weeks()
                    

            to_add.append((input.unit(), name, activity, role, variation, input.sick_leave(), input.on_campus(), input.units(), code, scheduled_start_time, scheduled_start, scheduled_weeks, unscheduled_flex, unscheduled_start, unscheduled_weeks))
        
        previous_df = current_df()
        to_add_df = pd.DataFrame(to_add, columns=previous_df.columns[1:])
        to_add_df['Delete'] = [get_delete_button(i) for i in range(rows_added(), rows_added() + to_add_df.shape[0])]
        to_add_df['RowID'] = [i for i in range(rows_added(), rows_added() + to_add_df.shape[0])]
        to_add_df.set_index('RowID', inplace=True)
        to_add_df = to_add_df[previous_df.columns]
        
        new_df = pd.concat([previous_df, to_add_df])
        current_df.set(new_df)
        rows_added.set(rows_added() + new_df.shape[0])

    @reactive.effect    
    @reactive.event(input.remove_row)
    def __():
        remove_ind = int(input.remove_row())
        new_df = render_df.data_view().drop(index=remove_ind)
        current_df.set(new_df)

    @render.data_frame
    def render_df():
        if input.add_button():
            return render.DataGrid(
                current_df(),
                editable=True,
                height=1000,
            )
        else:
            return render.DataGrid(
                base_df,
            )
        
    @render_df.set_patch_fn
    def _(*, patch: render.CellPatch) -> render.CellValue:
        if patch['column_index'] in [7]:
            return float(patch['value'])
        return patch['value']

    @render.download(filename=f'ota-{datetime.now().strftime("%Y-%m-%d-%H%M%S")}.xlsx')
    def download():
        current_df.set(render_df.data_view())        
        ota_path = os.path.join(os.path.dirname(__file__), 'base-ota.xlsx')
        new_path = os.path.join(os.path.dirname(__file__), f'ota-{datetime.now().strftime("%Y-%m-%d-%H%M%S")}.xlsx')
        shutil.copy(ota_path, new_path)
        def colour_phd(x, color):
            return np.where(np.isin(x, (staff_file()[staff_file()['PhDStatus'] == 'Y']).index), f'color: {color};', None)
        def border(x):
            return 'border: solid thin'
        with pd.ExcelWriter(new_path, mode='a', if_sheet_exists='overlay', engine='openpyxl') as writer:
            current_df().iloc[:, 1:].style.apply(colour_phd, color='red', subset='Staff').applymap(border).to_excel(writer, index=False, header=False, sheet_name='Sheet1', startrow=3)
        return new_path

    @render.download(filename='UOSCODE-staff-details.csv')
    def download_template():       
        path = os.path.join(os.path.dirname(__file__), 'TEST1234-staff-details.csv')
        return path
    
app = App(app_ui, server)
